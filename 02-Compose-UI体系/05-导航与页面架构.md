# 导航与页面架构

> Navigation Compose 是 Jetpack 的导航库，用于在 Composable 之间跳转。
> **后端类比**：Navigation ≈ Spring MVC 的路由映射。

---

## 1. 基础用法

```kotlin
// 定义路由（用字符串常亮避免拼写错误）
object Routes {
    const val HOME = "home"
    const val DETAIL = "detail/{newsId}"  // {newsId} 是路径参数
    fun detail(newsId: String) = "detail/$newsId"
}

// 设置导航
@Composable
fun AppNavGraph() {
    val navController = rememberNavController()

    NavHost(navController = navController, startDestination = Routes.HOME) {
        composable(Routes.HOME) {
            HomeScreen(onItemClick = { newsId ->
                navController.navigate(Routes.detail(newsId))
            })
        }
        composable(Routes.DETAIL) { backStackEntry ->
            val newsId = backStackEntry.arguments?.getString("newsId")
            DetailScreen(newsId = newsId ?: "")
        }
    }
}
```

**后端类比**：
- `NavHost` ≈ Spring 的 `DispatcherServlet` 或路由表
- `composable(route) { }` ≈ `@GetMapping("/path")`
- `navController.navigate()` ≈ `return "redirect:/path"`
- `backStackEntry.arguments` ≈ `@PathVariable`

## 2. 参数传递

```kotlin
// 方式一：路径参数（必传）
composable("detail/{newsId}") { entry ->
    val newsId = entry.arguments?.getString("newsId")
}

// 方式二：可选参数
composable("detail/{newsId}?source={source}") { entry ->
    val newsId = entry.arguments?.getString("newsId")
    val source = entry.arguments?.getString("source") ?: "unknown"
}

// 方式三：复杂对象（用 JSON 编码）
navController.navigate("detail/${Uri.encode(newsId)}")
```

**注意**：Navigation Compose 不建议传大对象。如果需要共享数据，用 ViewModel 的 Shared ViewModel 或数据库。

## 3. 返回和返回栈管理

```kotlin
// 返回上一页
navController.popBackStack()

// 返回并携带结果
navController.previousBackStackEntry?.savedStateHandle?.set("result", data)
navController.popBackStack()

// 在目标页面接收结果
val savedStateHandle = navController.currentBackStackEntry?.savedStateHandle
savedStateHandle?.getLiveData<String>("result")?.observe(...)

// 清空返回栈（如：登录后跳转主页，不应该能返回到登录页）
navController.navigate("home") {
    popUpTo("login") { inclusive = true }
}
```

## 5. 项目中的典型导航实现

```kotlin
// app/MainActivity.kt
class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContent {
            NewsAppTheme {
                AppNavGraph()
            }
        }
    }
}

// app/ui/navigation/NavGraph.kt
@Composable
fun AppNavGraph() {
    val navController = rememberNavController()

    NavHost(navController = navController, startDestination = "home") {
        composable("home") {
            HomeScreen(onArticleClick = { id ->
                navController.navigate("detail/$id")
            })
        }
        composable("detail/{id}") { entry ->
            DetailScreen(id = entry.arguments?.getString("id") ?: "")
        }
    }
}
```
